def all
  respond_to do |format|
    format.html do
      @all = Convocation.order(created_at: :desc)
      render layout: 'fylo'
    end
    format.xls
    format.pdf do
      pdf = PrintPdf.new
      send_data pdf.render, filename: "all_contravention.pdf", type: 'application/pdf', disposition: 'inline'
    end
  end

end
