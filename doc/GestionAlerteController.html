def archive
  id = params[:alerte_id]
  query = Alerte.find(id)
  query.archive = "archived"
  if query.save
    redirect_to(:back)
  else
    flash[:notice] = "Impossible d'archiver"
  end
end

def cartography
  if params[:id].nil
    id = params[:id].to_i #contient la date de debut
    @cartography = Alerte.where(created_at: id.days.ago..Date.today).order(created_at: :desc)
  else
    id = params[:id].to_i #contient la date de debut
